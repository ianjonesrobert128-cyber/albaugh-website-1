// Group products by base name
function getBaseName(p) {
  return p.replace(/\s*\(.*\)/, '').replace(/-\s*M\s*$/, '').trim().toUpperCase();
}

function buildGroups() {
  const map = {};
  rawProducts.forEach(p => {
    const base = getBaseName(p);
    if (!map[base]) map[base] = [];
    // avoid exact duplicates
    const clean = p.trim().replace(/\s*-\s*M\s*$/, '').trim();
    if (!map[base].find(x => x.toLowerCase() === clean.toLowerCase())) {
      map[base].push(clean);
    }
  });
  return map;
}

const groups = buildGroups();
const groupNames = Object.keys(groups).sort();
let colorMap = {};
groupNames.forEach((g, i) => { colorMap[g] = colorPalette[i % colorPalette.length]; });

// Populate filter
const sel = document.getElementById('groupFilter');
groupNames.forEach(g => {
  const o = document.createElement('option');
  o.value = g; o.textContent = g;
  sel.appendChild(o);
});

function getPackSize(p) {
  const m = p.match(/\(([^)]+)\)/);
  return m ? m[1] : '';
}

function openModal(name, pack) {
  document.getElementById('modal-title').textContent = name;
  document.getElementById('modal-pack').textContent = 'Pack: ' + pack;
  document.getElementById('modal').classList.add('open');
}
function closeModal() { document.getElementById('modal').classList.remove('open'); }

function renderProducts(filter = '', groupFilter = '') {
  const container = document.getElementById('products-container');
  container.innerHTML = '';
  let totalShown = 0;

  groupNames.forEach((gName, gi) => {
    if (groupFilter && gName !== groupFilter) return;
    const prods = groups[gName].filter(p => {
      if (!filter) return true;
      return p.toLowerCase().includes(filter.toLowerCase());
    });
    if (!prods.length) return;
    totalShown += prods.length;

    const col = colorMap[gName];
    const icon = icons[gi % icons.length];

    const section = document.createElement('div');
    section.className = 'product-group';

    const header = document.createElement('div');
    header.className = 'group-header';
    header.style.borderColor = col.border;
    header.innerHTML = `
      <div class="group-dot" style="background:${col.border}"></div>
      <div class="group-name" style="color:${col.text}">${gName}</div>
      <div class="group-count" style="background:${col.bg};color:${col.text};border:1px solid ${col.border}">${prods.length} variants</div>`;
    section.appendChild(header);

    const grid = document.createElement('div');
    grid.className = 'product-cards';

    prods.forEach(p => {
      const pack = getPackSize(p);
      const card = document.createElement('div');
      card.className = 'product-card';
      card.style.background = col.bg;
      card.style.borderColor = col.border;
      card.innerHTML = `
        <div class="pack-icon">${icon}</div>
        <div class="product-name" style="color:${col.text}">${gName}</div>
        <div class="product-pack" style="color:${col.text}">${pack}</div>`;
      card.onclick = () => openModal(gName, pack);
      grid.appendChild(card);
    });

    section.appendChild(grid);
    container.appendChild(section);
  });

  document.getElementById('no-results').style.display = totalShown === 0 ? 'block' : 'none';
  document.getElementById('total-count').textContent = totalShown + '+';
}

function filterProducts() {
  const q = document.getElementById('searchInput').value;
  const g = document.getElementById('groupFilter').value;
  renderProducts(q, g);
}
